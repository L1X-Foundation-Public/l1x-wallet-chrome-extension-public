get name

Example

        
        // JavaScript example
          
            window.L1X.makeRequestCall({
              contractAddress: "e08b886798933dfeaed6bf10e91ab4ba4bdee759",
              functionName: "nft_name",
              args: {}
            })
              .then((response) => console.log(response))
              .catch((err) => console.error(err));

mint nft token id to address

Example

        
        // JavaScript example
          
            const args = {
              to: "0xe74193e598fcf576cd099cee5b4ffe6efbea6b8e",
              id: "1"
            };
      
            window.L1X.sendTransaction({
              from: "0xe74193e598fcf576cd099cee5b4ffe6efbea6b8e",
              contractAddress: "e08b886798933dfeaed6bf10e91ab4ba4bdee759",
              functionName: "nft_mint_id_to",
              args: args,
              feeLimit: 100000
            })
              .then((response) => console.log(response))
              .catch((err) => console.error(err));
