get name

Example

        
        // JavaScript example
          
            window.L1X.makeRequestCall({
              contractAddress: "92157ef2c70f54547517e63b68b987d16f53444e",
              functionName: "ft_name",
              args: {}
            })
              .then((response) => console.log(response))
              .catch((err) => console.error(err));

mint

Example

        
        // JavaScript example
          
            const transactionData = {
              recipient_id: "0xfdf87120d9c9ef4be35fc474f9152ed62acfa69e",
              amount: 999
            };

            window.L1X.sendTransaction({
              from: "0xe74193e598fcf576cd099cee5b4ffe6efbea6b8e",
              contractAddress: "92157ef2c70f54547517e63b68b987d16f53444e",
              functionName: "ft_mint",
              args: transactionData,
              feeLimit: 1000000000000000000   // optional
            })
              .then((response) => {
                console.log("Transaction successful:", response);
              })
              .catch((error) => {
                console.error("Transaction failed:", error);
              });
